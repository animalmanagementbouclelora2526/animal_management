};

// ======================================================
// DECRYPT AES
// ======================================================

uint16_t decryptAES(byte* msg, uint16_t len) {

  byte iv[16] = {0};

  memset(cleartext, 0, sizeof(cleartext));

  return aesLib.decrypt(
      msg,
      len,
      (char*)cleartext,
      (byte*)aes_key,
      128,
      iv
  );
}


// ======================================================
// SAISIE KEYPAD
// # = VALIDER
// * = EFFACER
// ======================================================

String readKeypad(String title, int maxLen) {

  String input = "";

  display.clearDisplay();
  display.setCursor(0,0);
  display.println(title);
  display.println("");
  display.println("# = OK");
  display.println("* = CLEAR");
  display.display();

  while (true) {

    char key = keypad.getKey();

    if (key) {

      // ===== VALIDATION =====
      if (key == '#') {

        if (input.length() > 0) {

          return input;
        }
      }

      // ===== CLEAR =====
      else if (key == '*') {

        input = "";
      }

      // ===== AJOUT CARACTERE =====
      else {

        if (input.length() < maxLen) {

          input += key;
        }
      }

      // ===== AFFICHAGE =====
      display.clearDisplay();
      display.setCursor(0,0);

      display.println(title);
      display.println("");

      // masque password
      for (int i = 0; i < input.length(); i++) {
        display.print("*");
      }

      display.display();

      Serial.println(input);
    }
  }
}

int getLastAnimalId() {

  File file = SPIFFS.open("/animals.csv", FILE_READ);

  if(!file) return 0;

  int lastId = 0;

  // ignorer l'entête
  if(file.available())
    file.readStringUntil('\n');

  while(file.available()) {

    String line = file.readStringUntil('\n');
    line.trim();

    if(line.length() == 0)
      continue;

    int comma = line.indexOf(',');

    if(comma > 0) {

      int id =
        line.substring(0, comma).toInt();

      if(id > lastId)
        lastId = id;
    }
  }

  file.close();

  return lastId;
}

void saveDeletedAnimal(String mac, String status,String newFarmerID = "") {

  File file = SPIFFS.open("/animals.csv", FILE_READ);

  if(!file) {
    return;
  }

  File deleted = SPIFFS.open("/deleted.csv", FILE_APPEND);

  if(!deleted) {
    file.close();
    return;
  }

  file.readStringUntil('\n'); // header

  while(file.available()) {

    String line = file.readStringUntil('\n');
    line.trim();

    if(line.length() == 0)
      continue;

    String fields[13];;

    int start = 0;

    for(int i = 0; i < 12; i++) {

      int pos = line.indexOf(',', start);

      if(pos < 0)
        break;

      fields[i] = line.substring(start, pos);

      start = pos + 1;
    }

    fields[12] = line.substring(start);

    // MAC = colonne 1
    if(fields[1] == mac) {

      // Remplacer le statut par mort/vendu
      if(status == "mort") {

      fields[10] = "mort";

      }
      else if(status == "vendu") {

       fields[10] = "actif";        // l'animal reste actif
       fields[11] = newFarmerID;    // nouveau propriétaire

         }

      // Mettre à jour LastSync
      fields[12] = getDateTime();

      deleted.printf(
        "%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s\n",
        fields[0].c_str(),
        fields[1].c_str(),
        fields[2].c_str(),
        fields[3].c_str(),
        fields[4].c_str(),
        fields[5].c_str(),
        fields[6].c_str(),
        fields[7].c_str(),
        fields[8].c_str(),
        fields[9].c_str(),
        fields[10].c_str(),
        fields[11].c_str(),
        fields[12].c_str()
      );

      Serial.println("Animal sauvegarde dans deleted.csv");

      break;
    }
  }

  file.close();
  deleted.close();
}

bool deleteFromFile(String filename, String mac) {

  File file = SPIFFS.open(filename, FILE_READ);

  if(!file)
    return false;

  File temp = SPIFFS.open("/temp.csv", FILE_WRITE);

  bool found = false;

  while(file.available()) {

    String line = file.readStringUntil('\n');
    line.trim();

    if(line.length() == 0)
      continue;

    // conserver l'entête
    if(line.startsWith("ID,")) {

      temp.println(line);
      continue;
    }

    if(line.indexOf(mac) >= 0) {

      found = true;
      continue;
    }

    temp.println(line);
  }

  file.close();
  temp.close();

  SPIFFS.remove(filename);
  SPIFFS.rename("/temp.csv", filename);

  return found;
}

void deleteAnimal() {

  Serial.println("\n===== DELETE ANIMAL =====");

  Serial.println("Enter MAC :");
  String mac = readSerialInput();

  if(mac == "BACK")
    return;

  if(!animalExists(mac)) {

    Serial.println("Animal n'existe pas");

    display.clearDisplay();
    display.setCursor(0,0);
    display.println("Animal Not Found");
    display.display();

    return;
  }

  Serial.println("Status (mort/vendu) :");
  String statut = readSerialInput();

  if(statut == "BACK")
    return;


  // ===== ANIMAL VENDU =====
  if(statut == "vendu") {

    Serial.println("New Farmer ID :");
    String newFarmerID = readSerialInput();

    if(newFarmerID == "BACK")
      return;

    updateAnimalFarmerID(mac, newFarmerID);

    if(WiFi.status() == WL_CONNECTED) {

      syncAnimalByMac(mac);

    } else {

      saveDeletedAnimal(mac, "vendu", newFarmerID);

    }

    deleteFromFile("/animals.csv", mac);

    Serial.println("Animal vendu");

    return;
  }


  // ===== ANIMAL MORT =====
  if(statut == "mort") {

    updateAnimalStatus(mac, "mort");

    if(WiFi.status() == WL_CONNECTED) {

      syncAnimalByMac(mac);

    } else {

      saveDeletedAnimal(mac, "mort");

    }

    deleteFromFile("/animals.csv", mac);

    Serial.println("Animal mort");

    return;
  }

}

std::vector<String> receivedMacs;

void checkAnimals() {

  inventoryMode = true;

  int nbAnimals = getAnimalCount();

  Serial.println("\n===== CHECK ANIMALS =====");
  Serial.print("Nombre animaux : ");
  Serial.println(nbAnimals);

  unsigned long waitTime = nbAnimals * 400;

  Serial.print("Temps attente : ");
  Serial.print(waitTime);
  Serial.println(" ms");

  receivedMacs.clear();

  bool received = false;

  for(int attempt = 1; attempt <= 3; attempt++) {

    Serial.print("Tentative ");
    Serial.println(attempt);

    radio.transmit("DEMANDE");
    radio.startReceive();

    unsigned long startTime = millis();

    while(millis() - startTime < waitTime) {

      if(backPressed()) {
        inventoryMode = false;
  return;
}

      byte buf[128];

      int state = radio.receive(buf, sizeof(buf));

      if(state == RADIOLIB_ERR_NONE) {

        received = true;

        size_t len = radio.getPacketLength();

        uint16_t decLen =
          decryptAES(buf, len);

        if(decLen == 0)
          continue;

        cleartext[decLen] = '\0';

        String str =
          String((char*)cleartext);

        str.trim();

        Serial.print("Recu : ");
        Serial.println(str);

        // Extraction MAC
        int p0 = str.indexOf('|');
        int p1 = str.indexOf('|', p0 + 1);
        int p2 = str.indexOf('|', p1 + 1);

void compareAnimals() {

  std::vector<String> registeredMacs;
  std::vector<String> absentMacs;

  File file = SPIFFS.open("/animals.csv", FILE_READ);

  if(!file) {
    Serial.println("Erreur ouverture animals.csv");
    return;
  }

  // ignorer entête
  file.readStringUntil('\n');

  while(file.available()) {

    String line = file.readStringUntil('\n');
    line.trim();

    if(line.length() == 0)
      continue;

    int firstComma = line.indexOf(',');
    int secondComma = line.indexOf(',', firstComma + 1);

    if(firstComma < 0 || secondComma < 0)
      continue;

    String mac =
      line.substring(firstComma + 1, secondComma);

    mac.trim();

    registeredMacs.push_back(mac);
  }

  file.close();

  // rechercher absents
  for(auto &regMac : registeredMacs) {

    bool found = false;

    for(auto &rxMac : receivedMacs) {

      if(regMac == rxMac) {
        found = true;
        break;
      }
    }

    if(!found) {
      absentMacs.push_back(regMac);
    }
  }

  Serial.println("\n===== STATISTIQUES =====");

  Serial.print("Animaux enregistres : ");
  Serial.println(registeredMacs.size());

  Serial.print("Animaux presents : ");
  Serial.println(receivedMacs.size());

  Serial.print("Animaux absents : ");
  Serial.println(absentMacs.size());

  Serial.println("\n--- PRESENTS ---");

  for(auto &mac : receivedMacs) {
    Serial.println(mac);
  }

  Serial.println("\n--- ABSENTS ---");

  for(auto &mac : absentMacs) {
    Serial.println(mac);

  }
 
if(WiFi.status() == WL_CONNECTED) {

  for(auto &mac : receivedMacs) {

    syncAnimalByMac(mac);

    delay(200);
  }
}

String getDateTime() {

  struct tm timeinfo;

  if (!getLocalTime(&timeinfo)) {
    return "NO_TIME";
  }

  char buffer[25];

  strftime(
      buffer,
      sizeof(buffer),
      "%Y-%m-%d %H:%M:%S",
      &timeinfo
  );

  return String(buffer);
}

void connectWiFi() {

  Serial.println("Connexion WiFi...");

  WiFi.mode(WIFI_STA);
  WiFi.disconnect(true);
  delay(1000);

  WiFi.begin(ssid, passwordWifi);

  unsigned long start = millis();

  while (WiFi.status() != WL_CONNECTED &&
         millis() - start < 15000) {

    delay(500);
    Serial.print(".");
  }

  if (WiFi.status() == WL_CONNECTED) {

    Serial.println();
    Serial.println("WiFi Connecte");
    Serial.println(WiFi.localIP());

  } else {

    Serial.println();
    Serial.println("Echec WiFi");
  }
}

void syncTimeNTP() {

  Serial.println("Synchronisation NTP...");

  configTime(
      gmtOffset_sec,
      daylightOffset_sec,
      ntpServer
  );

  struct tm timeinfo;

  int retry = 0;

  while (!getLocalTime(&timeinfo) && retry < 20) {

    Serial.print(".");
    delay(500);
    retry++;
  }

  if (retry < 20) {

    Serial.println();
    Serial.println("Heure synchronisee");

    char buffer[30];

    strftime(
      buffer,
      sizeof(buffer),
      "%Y-%m-%d %H:%M:%S",
      &timeinfo
    );

    Serial.println(buffer);

  } else {

    Serial.println();
    Serial.println("Echec synchronisation NTP");
  }
}


void checkNtpSync() {

  if (WiFi.status() != WL_CONNECTED) {
    return;
  }

  if (millis() - lastNtpSync >= NTP_SYNC_INTERVAL) {

    Serial.println("Resynchronisation NTP...");

    configTime(
      gmtOffset_sec,
      daylightOffset_sec,
      ntpServer
    );

    struct tm timeinfo;

    if (getLocalTime(&timeinfo)) {

      lastNtpSync = millis();

      Serial.println("NTP synchronise");
      Serial.println(getDateTime());

    } else {

      Serial.println("Echec NTP");
    }
  }
}

         c=='-' ||
         c=='_' ||
         c=='.' ||
         c=='~') {

        encoded += c;
      }
      else if(c==' ') {

        encoded += '+';
      }
      else {

        encoded += '%';

        String temp =
          String(c, HEX);

        temp.toUpperCase();

        if(temp.length()==1)
          encoded += "0";

        encoded += temp;
      }

  display.setCursor(0,0);

  display.println("MAIN MENU");
  display.println("");

  for(int i = 0; i < menuCount; i++) {

  }

}

void checkButtons() {

  static unsigned long lastPress = 0;

  if(millis() - lastPress < 200)
    return;

  // ===== UP =====
  if(digitalRead(UP_BUTTON) == LOW) {

    lastPress = millis();

    if(inMenu) {

      currentMenuItem--;

      if(currentMenuItem < 0)
        currentMenuItem = menuCount - 1;

      displayMenu();
    }
  }

  // ===== DOWN =====
  if(digitalRead(DOWN_BUTTON) == LOW) {

    lastPress = millis();

    if(inMenu) {

      currentMenuItem++;

      if(currentMenuItem >= menuCount)
        currentMenuItem = 0;

      displayMenu();
    }
  }

  // ===== ENTER =====
  if(digitalRead(ENTER_BUTTON) == LOW) {

    lastPress = millis();

    if(inMenu) {

      inMenu = false;
      actionRunning = true;

      display.clearDisplay();
      display.setCursor(0,0);

      display.println("Selected:");
      display.println(menuItems[currentMenuItem]);

      display.display();

      delay(1000);
    }
  }

  // ===== BACK =====
if(digitalRead(BACK_BUTTON) == LOW) {

  lastPress = millis();

  inMenu = true;
  actionRunning = false;
  screenDisplayed = false;

  displayMenu();
}
}
